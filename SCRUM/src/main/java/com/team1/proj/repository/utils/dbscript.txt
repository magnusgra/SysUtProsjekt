DROP TABLE resultat;
DROP TABLE oppgave;
DROP TABLE bruker;

CREATE TABLE bruker
(
brukernavn VARCHAR(255),
epost VARCHAR(255),
rettigheter INT,
passord VARCHAR(255),
PRIMARY KEY(epost)
);

CREATE TABLE oppgave
(
oppgavenr INT,
PRIMARY KEY(oppgavenr)
);

CREATE TABLE resultat
(
brukernavn VARCHAR(255) NOT NULL,
forsoknr INT,
oppgavenr INT,
status VARCHAR(10),
poeng INT,
PRIMARY KEY(brukernavn,oppgavenr,forsoknr),
FOREIGN KEY(brukernavn) REFERENCES bruker(brukernavn),
FOREIGN KEY(oppgavenr) REFERENCES oppgave(oppgavenr)
);
